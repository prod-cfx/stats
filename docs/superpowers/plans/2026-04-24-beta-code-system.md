# 内测码系统 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a beta code gate that requires valid admin-generated beta codes only when a new user account is created, while keeping existing users' login flows unchanged.

**Architecture:** Add a focused backend `BetaCodeModule` with controller/service/repository boundaries, then integrate it into the existing auth creation paths through one service method. Extend OpenAPI contracts, user login UI, and the admin app so operators can generate, list, copy, enable, and disable beta codes.

**Tech Stack:** NestJS 11, Prisma 7, PostgreSQL, `nestjs-cls` transactions, Next.js 16, React 19, Ant Design, `@ai/api-contracts`, `dx` command runner.

```ts
import type { TransactionalAdapterPrisma } from '@nestjs-cls/transactional-adapter-prisma'
import type { BetaAccessCode } from '@/prisma/prisma.types'
import { TransactionHost } from '@nestjs-cls/transactional'
import { Injectable } from '@nestjs/common'

@Injectable()
export class BetaCodeRepository {
  constructor(private readonly txHost: TransactionHost<TransactionalAdapterPrisma>) {}

  async count(): Promise<number> {
    return this.txHost.tx.betaAccessCode.count()
  }

  async findMany(params: { skip: number; take: number }): Promise<BetaAccessCode[]> {
    return this.txHost.tx.betaAccessCode.findMany({
      orderBy: { createdAt: 'desc' },
      skip: params.skip,
      take: params.take,
    })
  }

  async createMany(codes: Array<{ code: string; maxUses: number; createdByAdminId?: string | null }>): Promise<BetaAccessCode[]> {
    return Promise.all(
      codes.map(item =>
        this.txHost.tx.betaAccessCode.create({
          data: {
            code: item.code,
            maxUses: item.maxUses,
            createdByAdminId: item.createdByAdminId ?? null,
          },
        }),
      ),
    )
  }

  async findByCode(code: string): Promise<BetaAccessCode | null> {
    return this.txHost.tx.betaAccessCode.findUnique({ where: { code } })
  }

  async incrementUsedCountIfAvailable(id: string): Promise<number> {
    const result = await this.txHost.tx.$executeRaw`
      UPDATE "beta_access_codes"
      SET "used_count" = "used_count" + 1,
          "updated_at" = NOW()
      WHERE "id" = ${id}
        AND "is_active" = true
        AND "used_count" < "max_uses"
    `
    return Number(result)
  }

  async createRedemption(data: { codeId: string; userId: string }): Promise<void> {
    await this.txHost.tx.betaAccessCodeRedemption.create({ data })
  }

  async updateUserInvitationCode(userId: string, code: string): Promise<void> {
    await this.txHost.tx.user.update({ where: { id: userId }, data: { invitationCode: code } })
  }

  async updateStatus(id: string, isActive: boolean): Promise<BetaAccessCode> {
    return this.txHost.tx.betaAccessCode.update({ where: { id }, data: { isActive } })
  }
}
```

- [ ] **Step 5: Create service**

Create `apps/backend/src/modules/beta-code/services/beta-code.service.ts`:

```ts
import { randomBytes } from 'node:crypto'
import { Injectable } from '@nestjs/common'
import { BasePaginationResponseDto } from '@/common/dto/base-pagination.response.dto'
import { isValidInvitationCode, normalizeInvitationCode } from '@ai/shared'
import { BetaCodeRepository } from '../repositories/beta-code.repository'
import {
  BetaCodeDisabledException,
  BetaCodeExhaustedException,
  BetaCodeInvalidException,
  BetaCodeRequiredException,
} from '../exceptions'

const DEFAULT_CODE_LENGTH_BYTES = 9

@Injectable()
export class BetaCodeService {
  constructor(private readonly repository: BetaCodeRepository) {}

  async list(params: { page: number; limit: number }) {
    const total = await this.repository.count()
    const items = await this.repository.findMany({
      skip: (params.page - 1) * params.limit,
      take: params.limit,
    })
    return new BasePaginationResponseDto(total, params.page, params.limit, items)
  }

  async createBatch(params: { count: number; maxUsesPerCode: number; adminId?: string | null }) {
    const codes = Array.from({ length: params.count }, () => ({
      code: this.generateCode(),
      maxUses: params.maxUsesPerCode,
      createdByAdminId: params.adminId ?? null,
    }))
    return this.repository.createMany(codes)
  }

  async updateStatus(id: string, isActive: boolean) {
    return this.repository.updateStatus(id, isActive)
  }

  async consumeForNewUser(params: { betaCode?: string | null; userId: string }): Promise<void> {
    const normalized = params.betaCode ? normalizeInvitationCode(params.betaCode).toUpperCase() : ''
    if (!normalized) throw new BetaCodeRequiredException()
    if (!isValidInvitationCode(normalized)) throw new BetaCodeInvalidException()

    const code = await this.repository.findByCode(normalized)
    if (!code) throw new BetaCodeInvalidException()
    if (!code.isActive) throw new BetaCodeDisabledException()
    if (code.usedCount >= code.maxUses) throw new BetaCodeExhaustedException()

    const updated = await this.repository.incrementUsedCountIfAvailable(code.id)
    if (updated !== 1) throw new BetaCodeExhaustedException()

    await this.repository.createRedemption({ codeId: code.id, userId: params.userId })
    await this.repository.updateUserInvitationCode(params.userId, code.code)
  }

  private generateCode(): string {
    return randomBytes(DEFAULT_CODE_LENGTH_BYTES)
      .toString('base64url')
      .replace(/[^A-Z0-9]/gi, '')
      .toUpperCase()
      .slice(0, 12)
  }
}
```

```tsx
'use client'

import type { ColumnsType } from 'antd/es/table'
import type { BetaCode } from '@/lib/api'
import { App, Button, Card, Form, InputNumber, Modal, Space, Switch, Table, Tag } from 'antd'
import { useCallback, useEffect, useMemo, useState } from 'react'
import { createBetaCodeBatch, fetchBetaCodes, updateBetaCodeStatus } from '@/lib/api'

interface BatchFormValues {
  count: number
  maxUsesPerCode: number
}

export default function BetaCodesPage() {
  const { message } = App.useApp()
  const [codes, setCodes] = useState<BetaCode[]>([])
  const [loading, setLoading] = useState(false)
  const [modalOpen, setModalOpen] = useState(false)
  const [generatedCodes, setGeneratedCodes] = useState<BetaCode[]>([])
  const [form] = Form.useForm<BatchFormValues>()

  const loadCodes = useCallback(async () => {
    setLoading(true)
    try {
      setCodes(await fetchBetaCodes())
    } catch (error: any) {
      message.error(error?.message ?? '获取内测码失败')
    } finally {
      setLoading(false)
    }
  }, [message])

  useEffect(() => {
    void loadCodes()
  }, [loadCodes])

  const handleCreate = async () => {
    const values = await form.validateFields()
    const created = await createBetaCodeBatch(values)
    setGeneratedCodes(created)
    message.success('内测码已生成')
    await loadCodes()
  }

  const copyGenerated = async () => {
    const text = generatedCodes.map(item => item.code).join('\n')
    await navigator.clipboard.writeText(text)
    message.success('已复制')
  }

  const columns: ColumnsType<BetaCode> = useMemo(() => [
    { title: '内测码', dataIndex: 'code' },
    {
      title: '状态',
      dataIndex: 'isActive',
      render: value => (value ? <Tag color="green">启用</Tag> : <Tag color="red">停用</Tag>),
    },
    {
      title: '使用量',
      render: (_, record) => `${record.usedCount}/${record.maxUses}`,
    },
    { title: '创建时间', dataIndex: 'createdAt' },
    {
      title: '操作',
      render: (_, record) => (
        <Switch
          checked={record.isActive}
          checkedChildren="启用"
          unCheckedChildren="停用"
          onChange={async checked => {
            await updateBetaCodeStatus(record.id, checked)
            await loadCodes()
          }}
        />
      ),
    },
  ], [loadCodes])

  return (
    <div className="page-container">
      <Card
        title="内测码"
        className="dashboard-card"
        extra={<Button type="primary" onClick={() => setModalOpen(true)}>批量生成</Button>}
      >
        <Table<BetaCode>
          rowKey="id"
          loading={loading}
          columns={columns}
          dataSource={codes}
          pagination={false}
        />
      </Card>

      <Modal
        title="批量生成内测码"
        open={modalOpen}
        onCancel={() => setModalOpen(false)}
        onOk={handleCreate}
        okText="生成"
      >
        <Space direction="vertical" size={16} style={{ width: '100%' }}>
          <Form layout="vertical" form={form} initialValues={{ count: 20, maxUsesPerCode: 1 }}>

              <InputNumber min={1} max={500} style={{ width: '100%' }} />
            </Form.Item>
            <Form.Item label="每个码可用次数" name="maxUsesPerCode" rules={[{ required: true, message: '请输入可用次数' }]}>
              <InputNumber min={1} max={1000} style={{ width: '100%' }} />
            </Form.Item>
          </Form>

          {generatedCodes.length > 0 && (
            <Card
              size="small"
              title="本批生成结果"
              extra={<Button size="small" onClick={copyGenerated}>复制</Button>}
            >
              <pre style={{ maxHeight: 240, overflow: 'auto', margin: 0 }}>
                {generatedCodes.map(item => item.code).join('\n')}
              </pre>
            </Card>
          )}
        </Space>
      </Modal>
    </div>
  )
}
```
